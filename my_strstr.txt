#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<assert.h>
char * my_strstr ( char * str, const char * substr) 
{
	int i = 0;
	int j = 0;
	int k = 0; 
	assert(str);
	assert(substr);
	for (i = 0; str[i] != '\0' ; i++) 
	{ 
		for (j = i, k = 0; (substr[k] != '\0')&&(str[j] == substr[k]); j++, k++) 
		{ 
			; 
		} 
		if (substr[k] == '\0'&&k > 0) 
			return &str[i]; 
	} 
	return NULL;
}
